## With close button

Add a close button to allow users to dismiss the feedback.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  preview={
    <Feedback
      title="Dismissible Feedback" 
      description="This feedback can be dismissed by clicking the close button."
      showCloseButton={true}
    />
  }
  codes={[
    {
      code:
`<Feedback
    title="Dismissible Feedback" 
    description="This feedback can be dismissed by clicking the close button."
    showCloseButton={true}
    onClose={() => {
      // Handle close action
      console.log('Feedback closed');
    }}
/>`,
      language: "tsx",
      label: "Close"
    }
  ]}
/>
